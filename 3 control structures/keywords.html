Keywords | continue | break | labels

  • Continue;

       Pasa a la siguiente iteración

  • Break;

       Rompe un ciclo

                        
    Usando el break y continue
    
                    for (let i=0; i<1000;i++){
                        if (false){
                            continue; //es mejor usar el continue con un if simple y no en un else
                        };
                        if (i>100){
                            break;
                        }else if(i%2===0){
                            console.log("Multiplo de 2:"+i);
                        }else if(i%3===0){
                            console.log("Multiplo de 3:"+i);
                        }else{
                            continue;//next iterartion
                            console.log("no interesa este número y esto nunca se va a imprimir")
                        };
                    };
